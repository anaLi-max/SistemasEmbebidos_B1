[7:31 p.m., 13/4/2026] Mi Numero: #include <Wire.h>
#include <SPI.h>
#include <SD.h>
#include <DHT.h>
#include <Adafruit_SSD1306.h>

// ── Pines ESP32 ─────────────────────────
#define PIN_DHT      4
#define PIN_CS_SD    5
#define PIN_BTN      2

#define PIN_LED_R    15
#define PIN_LED_G    2
#define PIN_LED_B    17

#define PIN_BUZZER   16

DHT dht(PIN_DHT, DHT22);
Adafruit_SSD1306 oled(128, 64, &Wire, -1);

// ── FSM ────────────────────────────────
enum Estado { MONITOR, ALERTA, LOG };
Estado estado_actual = MONITOR;

// ── Variables ─────────────────────────
float temp_c = 0, humedad = 0;
unsigned long t_ultimo_guardado = 0;
unsigned long t_btn_presionado  = 0;
int total_registros = 0;

const int INTERVALO_GUARDADO = 10000;

// ── Funciones ─────────────────────────
void setLED(bool r, bool g, bool b) {
  digitalWrite(PIN_LED_R, r);
  digitalWrite(PIN_LED_G, g);
  digitalWrite(PIN_LED_B, b);
}

void guardarEnSD() {
  File f = SD.open("datos.csv", FILE_APPEND);

  if (f) {
    f.print(millis()/1000); f.print(",");
    f.print(temp_c,1);     f.print(",");
    f.println(humedad,1);
    f.close();

    total_registros++;
    Serial.print("Guardado #");
    Serial.println(total_registros);
  }
}

// ── SETUP ─────────────────────────────
void setup() {
  Serial.begin(115200);

  pinMode(PIN_BTN, INPUT_PULLUP);
  pinMode(PIN_LED_R, OUTPUT);
  pinMode(PIN_LED_G, OUTPUT);
  pinMode(PIN_LED_B, OUTPUT);
  pinMode(PIN_BUZZER, OUTPUT);

  dht.begin();

  // I2C ESP32
  Wire.begin(21, 22);

  // OLED
  if(!oled.begin(SSD1306_SWITCHCAPVCC, 0x3C)) {
    Serial.println("Error OLED");
    while(true);
  }

  oled.clearDisplay();
  oled.setTextColor(SSD1306_WHITE);

  // SD
  if (SD.begin(PIN_CS_SD)) {
    if (!SD.exists("datos.csv")) {
      File f = SD.open("datos.csv", FILE_WRITE);
      f.println("tiempo_s,temp_c,humedad_pct");
      f.close();
    }
    Serial.println("SD lista.");
  } else {
    Serial.println("Error SD");
  }

  Serial.println("Sistema iniciado. Estado: MONITOR");
}

// ── LOOP ──────────────────────────────
void loop() {

  // Leer sensor
  float h = dht.readHumidity();
  float t = dht.readTemperature();

  if (!isnan(h) && !isnan(t)) {
    humedad = h;
    temp_c = t;
  }

  // Botón
  bool btn = (digitalRead(PIN_BTN) == LOW);

  if (btn) {
    if (t_btn_presionado == 0) t_btn_presionado = millis();
    if (millis() - t_btn_presionado > 3000) estado_actual = LOG;
  } else {
    t_btn_presionado = 0;
    if (estado_actual == LOG) estado_actual = MONITOR;
  }

  // Alerta
  if (temp_c > 35.0 || humedad > 85.0)
    estado_actual = ALERTA;
  else if (estado_actual == ALERTA)
    estado_actual = MONITOR;

  // OLED
  oled.clearDisplay();
  oled.setTextSize(1);

  switch (estado_actual) {

    case MONITOR:
      setLED(0,1,0);
      oled.setCursor(0,0); oled.print("MONITOR");

      oled.setTextSize(2);
      oled.setCursor(0,16);
      oled.print(temp_c,1); oled.print("C");

      oled.setTextSize(1);
      oled.setCursor(0,40);
      oled.print("Hum: "); oled.print(humedad,1); oled.print("%");

      oled.setCursor(0,54);
      oled.print("Reg: "); oled.print(total_registros);

      if (millis() - t_ultimo_guardado >= INTERVALO_GUARDADO) {
        guardarEnSD();
        t_ultimo_guardado = millis();
      }
    break;

    case ALERTA:
      setLED(1,0,0);
      tone(PIN_BUZZER, 1500, 300);

      oled.setCursor(20,0);
      oled.print("!!! ALERTA !!!");

      oled.setTextSize(2);
      oled.setCursor(0,16);
      oled.print(temp_c,1); oled.print("C");

      oled.setTextSize(1);
      oled.setCursor(0,42);

      if (temp_c > 35) oled.print("TEMP ALTA");
      if (humedad > 85) oled.print("HUM ALTA");
    break;

    case LOG:
      setLED(0,0,1);

      oled.setCursor(10,10);
      oled.print("MODO REGISTRO");

      oled.setCursor(10,28);
      oled.print("Total:");

      oled.setTextSize(2);
      oled.setCursor(10,40);
      oled.print(total_registros);

      oled.setTextSize(1);
      oled.setCursor(0,58);
      oled.print("Suelta boton");
    break;
  }

  oled.display();
  delay(500);
}
